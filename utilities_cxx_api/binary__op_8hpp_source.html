200 template<typename DerivedType, typename LHSType, typename RHSType>
201 template<typename DerivedType2, typename LHSType2, typename RHSType2>
202 bool BinaryOp<DerivedType, LHSType, RHSType>::operator==(
203  const BinaryOp<DerivedType2, LHSType2, RHSType2>& other) const noexcept {
204  using lhs2_type = typename TermTraits<LHSType2>::value_type;
205  using rhs2_type = typename TermTraits<RHSType2>::value_type;
206  constexpr auto l_same = std::is_same_v<lhs_type, lhs2_type>;
207  constexpr auto r_same = std::is_same_v<rhs_type, rhs2_type>;
208  if constexpr(l_same && r_same) {
209  return std::tie(lhs(), rhs()) == std::tie(other.lhs(), other.rhs());
210  } else {
211  return false;
212  }
213 }
