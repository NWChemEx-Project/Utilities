93  m_k_(k),
94  m_perm_([=]() {
95  const auto n = seq.size();
96  const auto eff_size = (n || k ? n + k - 1 : 0);
97  std::vector<bool> temp(!repeat ? n : eff_size, true);
98  for(std::size_t i = 0; i < k; ++i) temp[i] = false;

103 template<typename SequenceType>
104 const SequenceType& Combinations<SequenceType>::at_(size_type index) const {
105  m_comb_ = SequenceType(m_k_);
106  auto perm = m_perm_[index];
107  for(size_type i = 0, counter = 0, bar_count = 0; i < perm.size(); ++i) {
108  if(!perm[i])
109  m_comb_[counter++] = m_set_[!m_repeat_ ? i : bar_count];
110  else
111  ++bar_count;
112  if(counter == m_comb_.size()) // Early termination, rest are false
113  break;
114  }
