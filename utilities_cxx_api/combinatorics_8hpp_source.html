258 template<typename container_type>
259 std::size_t n_permutations(const container_type& seq) {
260  container_type temp(seq);
261  std::vector<std::size_t> counts;
262  counts.reserve(seq.size());
263  std::sort(temp.begin(), temp.end());
264  auto ei = temp.begin();
265  while(ei != temp.end()) {
266  const auto n_ei = std::count(ei, temp.end(), *ei);
267  counts.push_back(n_ei);
268  ei += n_ei;
269  }
270  return multinomial_coefficient<std::size_t>(counts);
271 }

274 template<typename container_type>
275 std::deque<std::size_t> permutation_to_fns(const container_type& perm,
276  const container_type& original) {
277  // Recursion end
278  if(perm.size() == 0) return std::deque<std::size_t>({});
279  if(perm.size() == 1) return std::deque<std::size_t>({0});
280 
281  // Strip off 1st element and put remaining elements in new container
282  auto perm_itr = perm.begin();
283  auto ei = *perm_itr;
284  container_type sub_perm(++perm_itr, perm.end());
285 
286  // Get digit in FNS and fill new_orig with all elements aside from it
287  std::size_t digit = 0;
288  container_type new_orig;
289  auto orig_itr = original.begin();
290  for(std::size_t i = 0; i < original.size(); ++i) {
291  auto ej = *orig_itr++;
292  if(ej == ei) {
293  digit = i;
294  while(orig_itr != original.end()) {
295  new_orig.insert(new_orig.end(), *orig_itr++);
296  }
297  break;
298  }
299  new_orig.insert(new_orig.end(), ej);
300  }
301  auto rv = permutation_to_fns(sub_perm, new_orig);
302  rv.push_front(digit);
303  return rv;
304 }

306 template<typename container_type>
307 container_type fns_to_permutation(const std::deque<std::size_t>& fns,
308  const container_type& original) {
309  container_type temp(original);
310  container_type rv;
311  for(auto x : fns) {
312  auto ei_itr = temp.begin();
313  container_type temp2;
314  for(std::size_t i = 0; i < temp.size(); ++i) {
315  if(i != x)
316  temp2.insert(temp2.end(), *ei_itr);
317  else
318  rv.insert(rv.end(), *ei_itr);
319  ++ei_itr;
320  }
321  temp.swap(temp2);
322  }
323  return rv;
324 }

326 template<typename container_type>
327 std::deque<std::size_t> fns_place_values(const container_type& perm) {
328  // Recursion end
329  if(perm.size() == 0) return std::deque<std::size_t>({});
330  if(perm.size() == 1) return std::deque<std::size_t>({1});
331  if(perm.size() == 2) return std::deque<std::size_t>({1, 1});
332 
333  // Strip off 1st element and put remaining elements in new container
334  auto perm_itr = perm.begin();
335  // auto ei=*perm_itr;
336  container_type sub_perm(++perm_itr, perm.end());
337 
338  auto rv = fns_place_values(sub_perm);
339  rv.push_front(n_permutations(perm) / perm.size());
340  return rv;
341 }

343 template<typename container_type>
344 std::deque<std::size_t> decimal_to_fns(std::size_t D,
345  const container_type& orig) {
346  container_type new_orig(orig.begin(), orig.end());
347  std::sort(new_orig.begin(), new_orig.end());
348  std::deque<std::size_t> rv;
349 
350  while(new_orig.begin() !=
351  new_orig.end()) // We pop an element out per iteration
352  {
353  auto curr_guess = new_orig.begin();
354  std::size_t total_perms = 0;
355  std::size_t digit_i = 0;
356  while(curr_guess != new_orig.end()) {
357  auto ei = *curr_guess;
358  // New container w/o element i
359  container_type temp(new_orig.begin(), curr_guess++);
360  temp.insert(temp.end(), curr_guess, new_orig.end());
361 
362  const auto P_ei = n_permutations(temp);
363  total_perms += P_ei;
364  if(D < total_perms) {
365  rv.push_back(digit_i);
366  new_orig.swap(temp);
367  D = D - (total_perms - P_ei);
368  break;
369  }
370  // We know the element isn't ei, but ei may still be in temp...
371  auto n_ei = std::count(temp.begin(), temp.end(), ei);
372  digit_i += 1 + n_ei;
373  std::advance(curr_guess, n_ei);
374  }
375  }
376 
377  return rv;
378 }
