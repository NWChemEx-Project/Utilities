127  constexpr bool operator<(const StaticString& rhs) const noexcept {
128  const bool am_smaller = (size() < rhs.size());
129  const std::size_t min_size = (am_smaller ? size_ : rhs.size());
130  for(std::size_t i = 0; i < min_size; ++i) {
131  const char li = (*this)[i];
132  const char ri = rhs[i];
133  if(li < ri)
134  return true;
135  else if(li > ri)
136  return false;
137  // must be that li == ri
138  }
139  // At this point either they are the same up the length of the shortest
140  // string. Thus we simply need to know which is shorter.
141  return am_smaller;
142  }
