102 template<typename SequenceType>
103 const SequenceType& Permutations<SequenceType>::at_(size_type i) const {
104  // We can quickly generate the current, next, and previous permutations, so
105  // pull those scenarios out of the general case.
106  if(i == m_counter_ + 1) {
107  std::next_permutation(m_buffer_.begin(), m_buffer_.end());
108  } else if(m_counter_ != 0 && i == m_counter_ - 1) {
109  std::prev_permutation(m_buffer_.begin(), m_buffer_.end());
110  } else if(i != m_counter_) {
111  m_buffer_ =
112  decimal_to_permutation((m_dx_ + i) % m_nperms_, m_sorted_orig_);
113  }
114  m_counter_ = i;
115  return m_buffer_;
116 }
