120 template<typename T>
121 class LeafHolder : public LeafHolderBase {
122 public:
124  static constexpr bool is_const = std::is_const_v<T>;
125 
127  using value_type = std::decay_t<T>;
128 
130  using const_reference = const value_type&;
131 
138  using reference =
139  std::conditional_t<is_const, const_reference, value_type&>;
140 
142  using const_pointer = const value_type*;
143 
150  using pointer = std::conditional_t<is_const, const_pointer, value_type*>;
151 
159  reference value() { return value_(); }
160 
170  const_reference value() const { return value_(); }
171 
172 protected:
173  LeafHolder() {
174  // This class shouldn't have references
175  static_assert(std::is_same_v<std::remove_reference_t<T>, T>);
176  }
177 
179  bool are_equal_(const LeafHolderBase& rhs) const noexcept override {
180  using mutable_type = LeafHolder<value_type>;
181  auto prhs = dynamic_cast<const mutable_type*>(&rhs);
182  if(prhs != nullptr) return value() == prhs->value();
183 
184  using immutable_type = LeafHolder<const value_type>;
185  auto pconst_rhs = dynamic_cast<const immutable_type*>(&rhs);
186  if(pconst_rhs == nullptr) return false;
187  return value() == pconst_rhs->value();
188  }
189 
191  bool contains_type_(bool should_be_const,
192  const rtti_type& type) const noexcept override {
193  return is_const == should_be_const && typeid(T) == type;
194  }
195 
196  bool is_convertible_(bool should_be_const,
197  const rtti_type& type) const noexcept override {
198  // If !should_be_const (i.e., wants mutable) then we must wrap mutable
199  if(!should_be_const) return contains_type_(false, type);
200 
201  // otherwise they want read-only and all that matters is the type.
202  return contains_type_(is_const, type);
203  }
204 
207  virtual reference value_() = 0;
208  virtual const_reference value_() const = 0;
210 };

222 private:

233  static constexpr bool is_rvalue = std::is_rvalue_reference_v<T>;
234 
236  static constexpr bool is_lvalue = std::is_lvalue_reference_v<T>;
237 
238 public:
241  using typename base_type::const_reference;
242  using typename base_type::leaf_pointer;
243  using typename base_type::pointer;
244  using typename base_type::reference;
245  using typename base_type::value_type;
247 
249  using holder_type = std::conditional_t<is_lvalue, pointer, value_type>;
250 
252  explicit QualifiedLeafHolder(T value) :
253  m_value_(make_holder_(std::forward<T>(value))) {}
254 
256  bool operator==(const QualifiedLeafHolder& rhs) const noexcept {
257  return m_value_ == rhs.m_value_;
258  }
259 
261  bool operator!=(const QualifiedLeafHolder& rhs) const noexcept {
262  return !((*this) == rhs);
263  }
264 
265 protected:
267  QualifiedLeafHolder() = delete;
268 
270  leaf_pointer clone_() const override {
271  return std::make_unique<my_type>(*this);
272  }
273 
275  reference value_() override { return unwrap_holder_(m_value_); }
276 
278  const_reference value_() const override { return unwrap_holder_(m_value_); }
279 
280 private:
282  static holder_type make_holder_(T object) {
283  if constexpr(is_lvalue) {
284  return pointer(&object);
285  } else {
286  return value_type(std::move(object));
287  }
288  }
289 
291  static reference unwrap_holder_(holder_type& value) {
292  if constexpr(is_lvalue) {
293  return *value;
294  } else {
295  return value;
296  }
297  }
298 
300  static const_reference unwrap_holder_(const holder_type& value) {
301  if constexpr(is_lvalue) {
302  return *value;
303  } else {
304  return value;
305  }
306  }
