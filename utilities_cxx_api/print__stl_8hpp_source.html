48 template<typename T>
49 std::ostream& print_element(std::ostream& os, T&& value) {
50  using clean_T = std::decay_t<T>;
51  if constexpr(type_traits::is_printable_v<clean_T>) {

79 template<typename T, typename Fxn>
80 std::ostream& print_queue(std::ostream& os, Fxn&& f, T&& rhs) {
81  os << '[';
82  bool printed = false;

141 template<typename T>
142 std::ostream& print_associative(std::ostream& os, T&& rhs, char odelim = '{',
143  char cdelim = '}') {
144  os << odelim;

149  os << "(";

180 template<std::size_t depth, typename T>
181 std::ostream& print_tuple(std::ostream& os, T&& rhs, char odelim = '(',
182  char cdelim = ')') {
183  constexpr auto max_depth = std::tuple_size_v<std::decay_t<T>>;
184  if constexpr(depth == max_depth && depth == 0) {
185  return os << odelim << cdelim;
186  } else if constexpr(depth == max_depth) {
187  return os << cdelim;
188  } else if constexpr(depth == 0) {
189  os << odelim;
190  print_element(os, std::get<0>(rhs));
191  return print_tuple<depth + 1>(os, std::forward<T>(rhs), odelim, cdelim);
192  } else {
193  os << ", ";
194  print_element(os, std::get<depth>(rhs));
195  return print_tuple<depth + 1>(os, std::forward<T>(rhs), odelim, cdelim);
196  }
197 }
