240 template<typename DerivedType>
241 bool operator==(const IndexableContainerBase<DerivedType>& lhs,
242  const IndexableContainerBase<DerivedType>& rhs) noexcept {
243  if(lhs.size() != rhs.size()) return false;
244  /* Note that here we don't use std::equal because it assumes that the return
245  * value of dereferencing an iterator is a true reference. Consequentially,
246  * we'll get segfaults if the user's container returns things by temporary
247  * copies.
248  */
249  for(decltype(lhs.size()) counter = 0; counter < lhs.size(); ++counter)
250  if(lhs[counter] != rhs[counter]) return false;
251  return true;
252 }

300  msg += std::to_string(index) + " is not in the range [0, ";
