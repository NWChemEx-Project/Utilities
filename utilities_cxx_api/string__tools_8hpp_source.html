44 inline std::vector<std::string> split_string(const std::string& str,
45  const std::string& delimiter) {
46  std::vector<std::string> strings;
47 
48  // Handle case that string is empty
49  if(str.empty()) return strings;
50 
51  std::string::size_type pos = 0;
52  std::string::size_type prev = 0;
53  while((pos = str.find(delimiter, prev)) != std::string::npos) {
54  strings.push_back(str.substr(prev, pos - prev));
55  prev = pos + delimiter.size();
56  }
57 
58  // We're out of delimiters, but haven't printed from the last delimiter to
59  // the end, so one more push_back
60  strings.push_back(str.substr(prev));
61 
62  return strings;
63 }
