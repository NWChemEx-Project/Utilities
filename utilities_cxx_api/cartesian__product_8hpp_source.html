74  template<typename IteratorType, std::size_t... I>
75  void run(const IteratorType& start, const IteratorType& end,
76  IteratorType& value, std::index_sequence<I...>) {
77  constexpr std::size_t nelems = sizeof...(I);
78  std::size_t idx = nelems;
79  while(true) {
80  if(!idx) break; // idx==0 means we have no indices left to try
81  // Increment the right most element that we know isn't at end
82  auto temp_value = tuple_transform(value, Incrementer(idx - 1));
83  // Now get first index at end
84  auto new_idx = tuple_find_if(temp_value, AtEnd<IteratorType>(end));
85  // If new_idx hasn't changed (i.e. is still idx) that was a good inc
86  if(new_idx == idx) {
87  value = tuple_transform(temp_value,
88  Reseter<IteratorType>(start, idx));
89  break;
90  }
91  idx = new_idx; // not good means new_idx has moved down one
92  }
93  }
94 };
